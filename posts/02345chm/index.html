def get_distance(X, Y, method='euclidean'):
    """
    è®¡ç®—å‘é‡Xã€Yçš„å„ç§è·ç¦»ï¼ŒXä¸Yå‡ä¸ºnp.arrayæ•°ç»„ï¼Œmethodä¸ºå­—ç¬¦ä¸²ï¼Œè¡¨ç¤ºè®¡ç®—å“ªç§è·ç¦»
    """
    if method == 'euclidean':  # æ¬§æ°è·ç¦»
        return np.linalg.norm(X - Y)
    elif method == 'cosine':  # ä½™å¼¦ç›¸ä¼¼åº¦
        return np.dot(X, Y)/(np.linalg.norm(X) * np.linalg.norm(Y))
    elif method == 'manhattan':  # æ›¼å“ˆé¡¿è·ç¦»
        return np.sum(np.abs(X - Y))
    elif method == 'chebyshev':  # åˆ‡æ¯”é›ªå¤«è·ç¦»
        return np.abs(X - Y).max()
    else:
        raise ValueError('æ²¡æœ‰æ­¤è·ç¦»å‚æ•°')


def knn_algorithm(test_X, train_X, train_Y, distance='euclidean', k: int=3):
